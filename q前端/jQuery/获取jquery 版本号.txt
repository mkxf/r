$.fn.jquery


Determine if an object is a jQuery object
 判断一个对象是不是jQuery 对象
var a = { what: "A regular JS object" },
b = $( "body" );

if ( a.jquery ) { // Falsy, since it's undefined
alert( "a is a jQuery object!" );
}

if ( b.jquery ) { // Truthy, since it's a string
alert( "b is a jQuery object!" );
}
